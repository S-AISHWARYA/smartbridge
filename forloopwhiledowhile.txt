USING FOR LOOP:

#include<stdio.h>
#include<conio.h>
main()
{
int i;
for(i=1;i<=120;i++)
{
printf("%d",i);
}
}

USING WHILE LOOP:

#include<stdio.h>
#include<conio.h>
main()
{ 
int i=1;
while(i<=120)
{
printf("%d",i);
i++;
}
}

USING  DO WHILE LOOP:

#include<stdio.h>
#include<conio.h>
main()
{
int i=1;
do
{ 
printf("%d",i);
i=i+1;
}while(i<=120);
}
